Find all the tables with a specific string in their name


SELECT 
    TABLE_SCHEMA, 
    TABLE_NAME
FROM 
    INFORMATION_SCHEMA.TABLES
WHERE 
    TABLE_NAME LIKE '%ErrorLog%';
